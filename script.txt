-- Fun√ß√£o Ninja para detectar o time (Busca por nome ou cor do time)
local function IdentificarCorDoTime(player)
    local team = player.Team
    local teamName = team and team.Name:lower() or ""
    local teamColor = team and team.TeamColor.Name or ""

    -- L√≥gica para o HOMER (Amarelo)
    if string.find(teamName, "homer") or teamColor:find("Yellow") or teamColor:find("Gold") then
        return Cores.Homer
    -- L√≥gica para o BART (Vermelho)
    elseif string.find(teamName, "bart") or teamColor:find("Red") or teamColor:find("Crimson") then
        return Cores.Bart
    end
    
    return Cores.Fallback
end

local function AdicionarESP(player)
    if player == LocalPlayer then return end

    local function aplicar(character)
        if not character then return end
        local root = character:WaitForChild("HumanoidRootPart", 10)
        local head = character:WaitForChild("Head", 10)
        if not root or not head then return end

        local corDoTime = IdentificarCorDoTime(player)

        -- 1. HIGHLIGHT (O brilho em volta do corpo)
        local highlight = character:FindFirstChild("HardcoreHighlight") or Instance.new("Highlight")
        highlight.Name = "HardcoreHighlight"
        highlight.Adornee = character
        highlight.FillColor = corDoTime
        highlight.OutlineColor = Color3.new(1, 1, 1)
        highlight.FillTransparency = 0.4
        highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        highlight.Parent = character

        -- 2. BILLBOARD GUI (O Nome flutuante)
        local billboard = head:FindFirstChild("NameESP") or Instance.new("BillboardGui")
        billboard.Name = "NameESP"
        billboard.Adornee = head
        billboard.Size = UDim2.new(0, 150, 0, 50)
        billboard.StudsOffset = Vector3.new(0, 3, 0)
        billboard.AlwaysOnTop = true
        billboard.Parent = head

        local label = billboard:FindFirstChild("TextLabel") or Instance.new("TextLabel")
        label.Parent = billboard
        label.BackgroundTransparency = 1
        label.Size = UDim2.new(1, 0, 1, 0)
        label.Text = (player.DisplayName or player.Name)
        label.TextColor3 = corDoTime -- Cor do texto igual a do time
        label.TextStrokeTransparency = 0
        label.TextSize = 16
        label.Font = Enum.Font.GothamBold
    end

    aplicar(player.Character)
    player.CharacterAdded:Connect(aplicar)
end

function AtivarESP()
    DesativarESP() -- Limpeza preventiva
    
    for _, player in pairs(Players:GetPlayers()) do
        AdicionarESP(player)
    end

    table.insert(Connections, Players.PlayerAdded:Connect(AdicionarESP))
    print("‚úÖ ESP COLORIDO ATIVADO: AMARELO (HOMER) | VERMELHO (BART)")
end

function DesativarESP()
    for _, conn in pairs(Connections) do if conn then conn:Disconnect() end end
    Connections = {}

    for _, player in pairs(Players:GetPlayers()) do
        local char = player.Character
        if char then
            local h = char:FindFirstChild("HardcoreHighlight")
            if h then h:Destroy() end
            if char:FindFirstChild("Head") then
                local n = char.Head:FindFirstChild("NameESP")
                if n then n:Destroy() end
            end
        end
    end
    print("‚ùå ESP DESATIVADO")
end
